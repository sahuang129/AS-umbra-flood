### 1. Pre-process the Binary Mask (Remove Noise & Smooth)
Before extracting contours, use morphological operations (like opening and closing) or Gaussian filtering to clean up the binary mask. This fills in small holes, removes isolated noise pixels, and naturally smooths the boundary.

```python
import numpy as np
from scipy import ndimage
from skimage import measure

def smooth_and_clean_mask(binary_mask, min_size=500, smooth_sigma=2.0):
    # Fill small holes in the water mask (e.g., small islands)
    cleaned_mask = ndimage.binary_fill_holes(binary_mask)
    
    # Remove small isolated water bodies (inland lakes/noise)
    # You can use morphological opening or simply remove small objects
    cleaned_mask = ndimage.binary_opening(cleaned_mask, iterations=3)
    
    # Apply a Gaussian filter to smooth the edges before contour extraction
    # This creates a "smooth surface" for the coastline
    smoothed_mask = ndimage.gaussian_filter(cleaned_mask.astype(float), sigma=smooth_sigma)
    
    # Re-binarize the smoothed mask (0.5 is the threshold)
    final_mask = smoothed_mask > 0.5
    
    return final_mask
```

### Method 1: Raster-Based Cropping (Pixel Coordinates)
If you are still working with pixel coordinates `(row, col)` before converting to geospatial coordinates, you can create a binary buffer mask from the reference landcover map and simply drop any points that fall outside of it.

If dropping points breaks the coastline into pieces, this function will automatically split it into separate valid segments:

```python
import numpy as np
from scipy import ndimage

def crop_coastline_to_raster_buffer(coastline_points, reference_landcover_mask, buffer_pixels=50):
    """
    Crops coastline points to a valid buffer zone derived from a reference landcover map.
    """
    # 1. Find the boundary of the reference landcover map
    # XOR the mask with its eroded self to get just the boundary edges
    eroded_mask = ndimage.binary_erosion(reference_landcover_mask)
    reference_edges = reference_landcover_mask ^ eroded_mask
    
    # 2. Dilate the boundary to create the valid buffer zone
    valid_zone = ndimage.binary_dilation(
        reference_edges, 
        structure=ndimage.generate_binary_structure(2, 1),
        iterations=buffer_pixels
    )
    
    # 3. Filter the extracted coastline points
    valid_segments = []
    current_segment = []
    
    for point in coastline_points:
        r, c = int(np.round(point[0])), int(np.round(point[1]))
        
        # Check if the point is within image bounds and inside the valid zone
        if (0 <= r < valid_zone.shape[0]) and (0 <= c < valid_zone.shape[1]) and valid_zone[r, c]:
            current_segment.append(point)
        else:
            # Point is out of bounds; cap off the current segment if it has enough points
            if len(current_segment) > 5: # min points to form a valid line
                valid_segments.append(np.array(current_segment))
            current_segment = [] # reset for the next valid stretch
            
    # Add the final segment if it ended inside the valid zone
    if len(current_segment) > 5:
        valid_segments.append(np.array(current_segment))
        
    return valid_segments
```

```python
from shapely.geometry import LineString, Polygon, MultiLineString

def crop_coastline_to_vector_buffer(coastline_line, reference_coastline_line, buffer_distance_meters):
    """
    Clips a Shapely LineString to a buffer zone of a reference LineString.
    Assumes coordinates are in a projected CRS (like UTM) where units are meters.
    """
    # 1. Create a buffer polygon around the reference landcover coastline
    valid_buffer_polygon = reference_coastline_line.buffer(buffer_distance_meters)
    
    # 2. Intersect the extracted coastline with the buffer polygon
    # This automatically crops out any parts of the line that extend beyond the buffer
    cropped_geometry = coastline_line.intersection(valid_buffer_polygon)
    
    # 3. Handle the output (intersection might return a LineString or a MultiLineString if split)
    if cropped_geometry.is_empty:
        return []
    elif isinstance(cropped_geometry, LineString):
        return [cropped_geometry]
    elif isinstance(cropped_geometry, MultiLineString):
        # Extract all the individual valid line segments
        return list(cropped_geometry.geoms)
    else:
        return []
```

### Which one should you use?

* **Use Method 2 (Vector)** if your reference landcover coastline comes from a shapefile (like GSHHG or OpenStreetMap), or if you want to specify your buffer distance in physical units (e.g., `buffer_distance_meters = 500`) rather than pixels.

```python
import numpy as np
import rasterio
from rasterio.warp import reproject, Resampling
from scipy import ndimage

def align_landcover_to_base(base_image_path, landcover_path):
    """
    Reprojects a landcover mask to exactly match the base image's resolution and grid.
    """
    with rasterio.open(base_image_path) as base_src:
        base_meta = base_src.profile
        base_transform = base_src.transform
        base_crs = base_src.crs
        base_shape = (base_src.height, base_src.width)
        
    with rasterio.open(landcover_path) as lc_src:
        # Read the landcover data (assuming land = 1, water = 0)
        lc_data = lc_src.read(1)
        
        # Create an empty array with the base image's dimensions
        aligned_landcover = np.zeros(base_shape, dtype=np.uint8)
        
        # Reproject the landcover data to match the base image
        reproject(
            source=lc_data,
            destination=aligned_landcover,
            src_transform=lc_src.transform,
            src_crs=lc_src.crs,
            dst_transform=base_transform,
            dst_crs=base_crs,
            resampling=Resampling.nearest # Use nearest neighbor for categorical/binary data
        )
        
    return aligned_landcover
```

### An Important Memory Optimization
While the previous code works, if your landcover mask is **significantly larger** than your SAR image (e.g., a continental or global landcover map), reading the entire landcover mask into memory using `lc_data = lc_src.read(1)` before reprojection can eat up a lot of RAM.

To make it highly efficient so that it **only reads the pixels it needs** from the large landcover file, you should use Rasterio's `WarpedVRT` (Virtual Raster Table). It performs the cropping, resolution change, and reprojection lazily on the fly.

Here is the optimized version of Step 1 for a larger landcover map:

```python
import numpy as np
import rasterio
from rasterio.vrt import WarpedVRT
from rasterio.enums import Resampling

def align_large_landcover_to_base(base_image_path, landcover_path):
    """
    Efficiently crops and reprojects a large landcover mask to match the base SAR image.
    Uses WarpedVRT to avoid reading the entire large landcover file into memory.
    """
    # 1. Get the spatial footprint and resolution of the SAR image
    with rasterio.open(base_image_path) as base_src:
        base_meta = base_src.profile
        base_transform = base_src.transform
        base_crs = base_src.crs
        base_height = base_src.height
        base_width = base_src.width

    # 2. Open the large landcover mask and lazily warp/crop it
    with rasterio.open(landcover_path) as lc_src:
        
        # Define the virtual raster matching the SAR image's grid
        vrt_options = {
            'crs': base_crs,
            'transform': base_transform,
            'height': base_height,
            'width': base_width,
            'resampling': Resampling.nearest
        }
        
        with WarpedVRT(lc_src, **vrt_options) as vrt:
            # This only reads and processes the small chunk that overlaps the SAR image!
            aligned_landcover = vrt.read(1)
            
    return aligned_landcover
```

### Why this is the best approach for large masks:
1. **Memory Safe:** If your landcover map is 50GB, this script will still only use a few megabytes of RAM because it only reads the bounding box that intersects your SAR image.
2. **Fast:** It skips processing any pixels outside your area of interest.
3. **Perfect Alignment:** The output array (`aligned_landcover`) will have the exact same number of rows and columns as your SAR image, meaning `aligned_landcover[row, col]` maps perfectly to `sar_image[row, col]`.

>> ChatGSFC:
**Yes, it is absolutely possible.** In SAR imagery, geometric distortions like **layover** (where mountain tops appear to lean toward the sensor) and **shadow** (where terrain blocks the radar beam) cause severe false detections, especially near coastal cliffs or steep fjords. 

Because radar is a side-looking instrument, these distortions are highly predictable if you know the terrain elevation and the sensor's viewing angle.

Here is how you can identify and remove these regions from your coastline detection pipeline:

### 1. The Best Approach: Generate a Layover/Shadow Mask (LSM)
To identify these regions accurately, you need a **Digital Elevation Model (DEM)** and the **SAR sensor metadata** (orbit and look angle). 

Instead of writing complex trigonometric formulas from scratch, the standard practice is to use SAR processing software to generate a **Layover/Shadow Mask (LSM)** during the Terrain Correction phase. 
* If you use **ESA SNAP** (Sentinel Application Platform), when you run "Range Doppler Terrain Correction", you can check the box for **"Output layover/shadow mask"**.
* This outputs a raster (GeoTIFF) where pixels are classified as:
  * `0`: Valid data
  * `1`: Shadow
  * `2`: Layover

### 2. Python Approximation: Local Incidence Angle (LIA)
If you don't have an LSM but you have a DEM and the SAR incidence angle (often found in the metadata or `gdalinfo`), you can calculate the **Local Incidence Angle (LIA)**. 
* **Shadow** occurs where the terrain slope facing away from the radar is steeper than the radar beam's depression angle.
* **Layover** occurs where the terrain slope facing the radar is steeper than the radar beam's incidence angle.

You can calculate the slope and aspect from the DEM using `scipy` or `richdem`, and flag pixels where the geometry causes these physical impossibilities.

### 3. How to remove them from your coastline
Once you have generated or obtained a Layover/Shadow mask (where `1` means distorted and `0` means valid), you can remove the distorted regions using the exact same logic we used for the landcover buffer.

Since the LSM mask will be on the same grid as your SAR image, you can simply drop coastline points that fall inside the distorted pixels:

```python
import numpy as np

def filter_coastline_by_distortion_mask(coastline_lines, distortion_mask):
    """
    Removes segments of the coastline that fall inside geometrically distorted regions.
    
    Parameters:
    - coastline_lines: List of (N, 2) numpy arrays [row, col]
    - distortion_mask: 2D numpy array where True/1 indicates Layover/Shadow/Invalid.
    """
    valid_lines = []
    
    for line in coastline_lines:
        current_segment = []
        
        for point in line:
            r, c = int(np.round(point[0])), int(np.round(point[1]))

            if (0 <= r < distortion_mask.shape[0]) and (0 <= c < distortion_mask.shape[1]):
                
                # If the pixel is NOT in a distorted region (e.g., mask == 0)
                if not distortion_mask[r, c]:
                    current_segment.append(point)

                    # We hit a distorted pixel (layover/shadow), break the coastline

                    current_segment = []
                    
        # Catch the final segment

            valid_lines.append(np.array(current_segment))
            
    return valid_lines
```

When a coastline contour loops back on itself in a 180-degree direction, it is usually tracing around a narrow feature like a river mouth, a narrow inlet, a breakwater, or a pier. 

There are two ways to fix this: **prevent it in the raster mask before contouring**, or **cut the loop out of the extracted line**. 

Here are the two best methods to handle this.

### Method 1: Prevent Loops using Morphology (Recommended & Easiest)
The most robust way to stop the contour from traveling up narrow rivers or around narrow piers is to "bridge" these gaps in your binary water mask *before* extracting the contours. 

You can use **Morphological Opening and Closing**:
* **Closing the water mask** fills in narrow land features (like piers or breakwaters).
* **Opening the water mask** removes narrow water features (like rivers or inlets).

```python
from scipy import ndimage

def bridge_narrow_features(binary_water_mask, gap_size_pixels=10):
    """
    Prevents contours from looping into narrow inlets or around narrow piers.
    """
    # Create a structuring element (a circle/disk) based on how wide of a gap you want to bridge
    struct_size = int(gap_size_pixels)
    y, x = np.ogrid[-struct_size:struct_size+1, -struct_size:struct_size+1]
    disk = x**2 + y**2 <= struct_size**2
    
    # 1. Morphological Opening (Erode then Dilate)
    # Removes narrow rivers/inlets so the contour goes straight across the mouth
    processed_mask = ndimage.binary_opening(binary_water_mask, structure=disk)
    
    # 2. Morphological Closing (Dilate then Erode)
    # Fills in narrow piers/spits so the contour ignores them
    processed_mask = ndimage.binary_closing(processed_mask, structure=disk)
    
    return processed_mask
```
*Run this function on your water mask just before you run `skimage.measure.find_contours`.*

### Method 2: "Short-Circuit" the Extracted Line (Vector Approach)
If you already have the extracted coastline as a list of coordinates and want to snip out the loops, you can use a "path distance vs. Euclidean distance" algorithm. 

If two points on the line are physically very close to each other (e.g., 5 pixels apart), but are very far apart along the sequence of the line (e.g., 100 points apart), it means the line took a massive detour (a loop) and came back. This script detects that and draws a straight line across the gap, discarding the loop.

```python
import numpy as np

def remove_hairpin_loops(coastline_line, spatial_tolerance=5.0, index_tolerance=20):
    """
    Removes 180-degree loops by short-circuiting points that are physically close 
    but far apart along the contour path.
    
    Parameters:
    - coastline_line: (N, 2) numpy array of [row, col] coordinates
    - spatial_tolerance: Max distance in pixels to consider two points "touching"
    - index_tolerance: Min number of points in the loop to consider it worth cutting
    """
    if len(coastline_line) < index_tolerance:
        return coastline_line
        
    keep_indices = [0]
    curr_idx = 0
    num_points = len(coastline_line)
    
    while curr_idx < num_points - 1:
        # Calculate straight-line distances from the current point to ALL future points
        dists = np.linalg.norm(coastline_line[curr_idx+1:] - coastline_line[curr_idx], axis=1)
        
        # Find points that are physically close
        close_points = np.where(dists < spatial_tolerance)[0]
        
        jump_made = False
        # Look backwards through the close points to find the LARGEST jump we can make
        for j in reversed(close_points):
            actual_index = curr_idx + 1 + j
            
            # If the jump skips a large number of points, it's a loop!
            if (actual_index - curr_idx) > index_tolerance:
                curr_idx = actual_index # Jump across the gap (short-circuit)
                keep_indices.append(curr_idx)
                jump_made = True
                break
                
        # If no loop was detected, just move to the very next point
        if not jump_made:
            curr_idx += 1
            keep_indices.append(curr_idx)
            
    return coastline_line[keep_indices]
```

### Which should you use?
* Use **Method 1** if the loops are caused by actual geography (like rivers or inlets) or noise in your SAR classification. It creates a much smoother, more natural "outer boundary."
* Use **Method 2** if you want strict control over processing the geometry *after* you have already applied your Landcover and Layover/Shadow masks.
