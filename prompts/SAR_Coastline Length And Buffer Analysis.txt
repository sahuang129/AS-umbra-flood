### Important Note on Coordinate Reference Systems (CRS)
To accurately compute lengths and buffers in **meters**, your shapefiles and raster must be in a projected CRS (like UTM) rather than a geographic CRS (like WGS84 / EPSG:4326, which uses degrees). The script includes a check to reproject if necessary, but you should update `TARGET_CRS` to the appropriate local EPSG code for your region.

### Python Script

```python
import os
import glob
import numpy as np
import geopandas as gpd
import rasterio
from rasterio.mask import mask
from shapely.geometry import box

# --- Configuration ---
SHP_DIR = "/path/to/your/shapefiles"
REF_RASTER_PATH = "/path/to/reference_land_water_mask.tif"
# Define a projected CRS (e.g., a UTM zone) to ensure lengths and buffers are in meters
TARGET_CRS = "EPSG:32618"  # Change this to your local UTM EPSG code!
BUFFER_DISTANCES = [1, 5, 10] # in meters

def process_coastlines():
    # Find all shapefiles in the directory
    shp_files = glob.glob(os.path.join(SHP_DIR, "*.shp"))
    
    total_coastline_length = 0.0
    buffer_stats = {dist: {'land_pixels': 0, 'water_pixels': 0} for dist in BUFFER_DISTANCES}
    
    # Open the reference raster once
    with rasterio.open(REF_RASTER_PATH) as src:
        raster_crs = src.crs
        
        for shp_file in shp_files:
            print(f"Processing {os.path.basename(shp_file)}...")
            
            # 1. Load shapefile
            gdf = gpd.read_file(shp_file)
            
            # Ensure it's in the target projected CRS for accurate meter measurements
            if gdf.crs != TARGET_CRS:
                gdf = gdf.to_crs(TARGET_CRS)
                
            # 2. Compute Length
            # Length of multilinestrings/linestrings in the CRS unit (meters)
            length = gdf.geometry.length.sum()
            total_coastline_length += length
            print(f"  Length: {length:.2f} m")
            
            # 3. Buffer Analysis
            # We must project the geometries to match the raster's CRS before clipping
            gdf_raster_crs = gdf.to_crs(raster_crs)
            
            for dist in BUFFER_DISTANCES:
                # Create buffer around the coastlines in the projected CRS (meters)
                # Then project the buffer to the raster's CRS to extract the pixels
                buffered_geoms_target_crs = gdf.geometry.buffer(dist)
                buffered_gdf = gpd.GeoDataFrame(geometry=buffered_geoms_target_crs, crs=TARGET_CRS)
                buffered_gdf_raster_crs = buffered_gdf.to_crs(raster_crs)
                
                # Merge buffers to prevent overlapping pixel double-counting
                merged_buffer = buffered_gdf_raster_crs.unary_union
                
                # Convert the merged shapely geometry to a format rasterio expects
                geom_for_mask = [merged_buffer] if merged_buffer.geom_type == 'Polygon' else list(merged_buffer.geoms)

                    # crop=True clips the raster extent to the bounding box of the shapefile/buffer
                    out_image, out_transform = mask(src, geom_for_mask, crop=True)
                    
                    # Assuming raster has a single band. Extract it.
                    out_band = out_image[0]
                    
                    # Analyze raster pixels within the buffer
                    # Replace 1 and 0 with your actual raster land/water classification values
                    # Usually, land=1, water=0, and nodata is something else (e.g., 255 or negative)
                    nodata = src.nodata if src.nodata is not None else -9999
                    
                    valid_pixels = out_band[out_band != nodata]
                    
                    # Example: counting occurrences of specific pixel values
                    # Update these condition checks based on your specific raster encoding!
                    land_count = np.sum(valid_pixels == 1)
                    water_count = np.sum(valid_pixels == 0)

                    buffer_stats[dist]['water_pixels'] += water_count
                    
                except ValueError as e:

                    print(f"  Skipping {dist}m buffer mask: {e}")

    print("\n" + "="*40)
    print("FINAL RESULTS")
    print("="*40)

    print("\nBuffer Analysis Results (Aggregated):")
    for dist in BUFFER_DISTANCES:
```

### How this script works:
1. **Length Calculation**: The geometries are reprojected into a projected CRS where units are in meters (e.g. UTM). It adds the `gdf.geometry.length.sum()` to a running total.
2. **Buffer Creation**: Within the UTM CRS, it applies `.buffer(dist)` for 1, 5, and 10 meters. It then dissolves (merges) these polygons using `.unary_union` so that areas where the buffer overlaps itself don't double-count raster pixels.
3. **Raster Clipping (`rasterio.mask`)**: It takes the reference raster and directly extracts *only* the pixels that fall inside those buffer bounds, safely clipping the large mask down to the immediate region surrounding the local shapefile.
4. **Analysis**: It separates the valid pixels extracted from the buffer and calculates how many land vs. water pixels fall inside that threshold. **Note**: You will need to modify the `1` and `0` logic under the `valid_pixels` checks depending on what integer values your reference mask uses for Land vs. Water.

```python
import os
import glob
import geopandas as gpd
import rasterio
from rasterio.mask import mask
from rasterio.features import shapes
from shapely.geometry import shape, box

# --- Configuration ---
SHP_DIR = "/path/to/your/shapefiles"
REF_RASTER_PATH = "/path/to/reference_land_water_mask.tif"

# Define a projected CRS (e.g., a UTM zone) to ensure lengths and buffers are in meters
TARGET_CRS = "EPSG:32618"  # Change this to your local UTM EPSG code!
BUFFER_DISTANCES = [1, 5, 10] # in meters

# Define your raster values (Update these to match your raster's encoding)
LAND_VALUE = 1  
WATER_VALUE = 0 

def process_coastlines():
    shp_files = glob.glob(os.path.join(SHP_DIR, "*.shp"))
    
    total_sar_length = 0.0
    buffer_stats = {dist: 0.0 for dist in BUFFER_DISTANCES} # Will store sum of intersecting lengths
    
    with rasterio.open(REF_RASTER_PATH) as src:
        raster_crs = src.crs
        
        for shp_file in shp_files:
            print(f"\nProcessing {os.path.basename(shp_file)}...")
            
            # 1. Load SAR coastline shapefile
            gdf = gpd.read_file(shp_file)
            
            # Ensure it's in the target projected CRS for accurate meter measurements
            if gdf.crs != TARGET_CRS:
                gdf = gdf.to_crs(TARGET_CRS)
                
            # Compute Length of this specific shapefile
            shp_length = gdf.geometry.length.sum()
            total_sar_length += shp_length
            print(f"  SAR Coastline Length: {shp_length:.2f} m")
            
            # 2. Define a bounding box to clip the raster
            # We add a generous 50m buffer to the bounding box to ensure we capture 
            # reference land/water interfaces that might be just outside the SAR line
            bbox_geom = box(*gdf.total_bounds).buffer(50) 
            bbox_gdf_raster_crs = gpd.GeoDataFrame(geometry=[bbox_geom], crs=TARGET_CRS).to_crs(raster_crs)
            
            try:
                # 3. Clip the raster to the expanded bounding box of the shapefile
                out_image, out_transform = mask(src, bbox_gdf_raster_crs.geometry, crop=True)
                out_band = out_image[0]
                
                # 4. Extract the Land/Water Interface (edges) from the raster
                # Vectorize the clipped raster
                results = (
                    {'properties': {'val': v}, 'geometry': s}
                    for i, (s, v) in enumerate(shapes(out_band, transform=out_transform))
                )
                
                # Convert to GeoDataFrame
                geoms = list(results)
                if not geoms:
                    print("  No raster data found in this extent.")
                    continue
                    
                raster_gdf = gpd.GeoDataFrame.from_features(geoms, crs=raster_crs)
                
                # Filter for land polygons and get their boundaries (the interface)
                # Note: getting the boundary of land inherently gives us the land/water edge
                land_polygons = raster_gdf[raster_gdf['val'] == LAND_VALUE]
                
                if land_polygons.empty:
                    print("  No land found in reference raster for this extent.")
                    continue
                    
                # Convert polygons to boundaries (lines) and project to meters
                ref_interface = land_polygons.geometry.boundary
                ref_interface_meters = ref_interface.to_crs(TARGET_CRS).unary_union
                
                # 5. Buffer Analysis (Intersection)
                # Combine all SAR lines in this shapefile into one geometry for cleaner intersection
                sar_lines_union = gdf.unary_union 
                
                for dist in BUFFER_DISTANCES:
                    # Buffer the reference mask's interface
                    ref_buffer = ref_interface_meters.buffer(dist)
                    
                    # Intersect SAR coastline with the reference buffer
                    intersecting_sar = sar_lines_union.intersection(ref_buffer)
                    
                    # Calculate the length of the SAR line that falls inside this buffer
                    intersecting_length = intersecting_sar.length
                    buffer_stats[dist] += intersecting_length
                    
                    percent_in_buffer = (intersecting_length / shp_length) * 100 if shp_length > 0 else 0
                    print(f"  {dist}m Buffer: {percent_in_buffer:.2f}% ({intersecting_length:.2f} m)")
                    
            except ValueError as e:
                print(f"  Skipping raster clip (likely falls outside raster extent): {e}")

    # --- Final Output ---
    print("\n" + "="*50)
    print("FINAL AGGREGATED RESULTS")
    print("="*50)
    print(f"Total files processed: {len(shp_files)}")
    print(f"Collective SAR Coastline Length: {total_sar_length:.2f} meters ({total_sar_length/1000:.2f} km)\n")
    
    print("Percentage of Total SAR Coastline within Reference Land/Water Interface:")
    for dist in BUFFER_DISTANCES:
        total_intersecting = buffer_stats[dist]
        overall_percent = (total_intersecting / total_sar_length) * 100 if total_sar_length > 0 else 0
        print(f"  {dist}m Buffer: {overall_percent:.2f}% (Total matching length: {total_intersecting:.2f} m)")

if __name__ == "__main__":
    process_coastlines()
```

### Key Logic Updates:
1. **Bounding Box Clipping**: It gets the total bounds of the current SAR shapefile, expands it by 50 meters, projects that bounding box to the raster's CRS, and uses it to clip the massive reference raster down to just the local area.
2. **Vectorizing the Interface (`rasterio.features.shapes`)**: It takes the local clipped raster array and turns contiguous pixel values into Polygons. It filters for the `LAND_VALUE` and converts the polygons to `LineStrings` using `.boundary`. This isolates the exact reference land/water edge.
3. **Buffering the Reference (Not the SAR)**: It creates 1m, 5m, and 10m buffers around the *reference edge*.
4. **Length Intersection**: It uses `.intersection()` to isolate the exact segments of your SAR shapefile that fall inside those reference buffers, calculates their length, and computes the percentage on both a per-file and a global collective basis.
