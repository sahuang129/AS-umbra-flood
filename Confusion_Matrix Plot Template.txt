Here is a clean, highly customizable Python template for plotting a confusion matrix. It uses **scikit-learn** to compute the matrix and **seaborn** with **matplotlib** to create a professional-looking heatmap.

### Python Template

```python
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
from sklearn.metrics import confusion_matrix

def plot_confusion_matrix(y_true, y_pred, class_names, title='Confusion Matrix', cmap='Blues'):
    """
    Plots a confusion matrix heatmap.
    
    Parameters:
    - y_true: Ground truth (correct) target values
    - y_pred: Estimated targets as returned by a classifier
    - class_names: List of string names for the classes
    - title: Title of the plot
    - cmap: Matplotlib colormap (e.g., 'Blues', 'viridis', 'crest')
    """
    # 1. Compute the confusion matrix
    cm = confusion_matrix(y_true, y_pred)
    
    # 2. Set up the matplotlib figure
    plt.figure(figsize=(8, 6))
    
    # 3. Create the heatmap using seaborn
    # fmt='d' ensures numbers are formatted as integers (use '.2f' if normalizing)
    sns.heatmap(cm, annot=True, fmt='d', cmap=cmap, cbar=True,
                xticklabels=class_names, yticklabels=class_names,
                annot_kws={"size": 12}) # Adjust font size of annotations
    
    # 4. Add labels and title
    plt.title(title, fontsize=16, pad=15)
    plt.ylabel('True Label', fontsize=12, fontweight='bold')
    plt.xlabel('Predicted Label', fontsize=12, fontweight='bold')
    
    # Rotate tick labels for better readability if class names are long
    plt.xticks(rotation=45, ha='right')
    plt.yticks(rotation=0)
    
    # 5. Adjust layout and show
    plt.tight_layout()
    plt.show()

# ==========================================
# Example Usage:
# ==========================================
if __name__ == "__main__":
    # Sample actual and predicted labels
    y_true_sample = [0, 1, 0, 1, 0, 1, 2, 2, 2, 0, 1, 2]
    y_pred_sample = [0, 1, 0, 2, 0, 1, 2, 1, 2, 0, 1, 1]
    
    # Define the names of your classes in order (0, 1, 2)
    labels = ['Healthy', 'Degraded', 'Critical']
    
    # Generate the plot
    plot_confusion_matrix(y_true_sample, y_pred_sample, class_names=labels)
```

>> ChatGSFC:
To compute a confusion matrix from two shapefiles, you need to spatially align the data so you can compare the "True" labels from one shapefile against the "Predicted" labels from the other. 

The most robust way to do this in Python is using **GeoPandas** to perform a spatial join (e.g., checking which predicted polygon a ground truth point falls into), and then feeding those matched columns into the scikit-learn confusion matrix function.

Here is a complete template:

### Prerequisites
You will need `geopandas` installed alongside the other libraries:
```bash
pip install geopandas numpy matplotlib seaborn scikit-learn
```
*(Note: geopandas relies on spatial indexing, so you may also need `rtree` or `pygeos` depending on your environment).*

### Python Template

```python
import geopandas as gpd
import matplotlib.pyplot as plt
import seaborn as sns
from sklearn.metrics import confusion_matrix
import numpy as np

def plot_confusion_matrix(y_true, y_pred, class_names, title='Confusion Matrix', cmap='Blues'):
    """Plots a confusion matrix heatmap."""
    cm = confusion_matrix(y_true, y_pred, labels=class_names)
    
    plt.figure(figsize=(8, 6))
    sns.heatmap(cm, annot=True, fmt='d', cmap=cmap, cbar=True,
                xticklabels=class_names, yticklabels=class_names)
    
    plt.title(title, fontsize=16, pad=15)
    plt.ylabel('True Label (Ground Truth)', fontsize=12, fontweight='bold')
    plt.xlabel('Predicted Label (Classification)', fontsize=12, fontweight='bold')
    plt.xticks(rotation=45, ha='right')
    plt.yticks(rotation=0)
    plt.tight_layout()
    plt.show()

def shapefiles_to_confusion_matrix(truth_shp_path, pred_shp_path, truth_col, pred_col):
    """
    Loads two shapefiles, aligns them spatially, and plots a confusion matrix.
    
    Parameters:
    - truth_shp_path: Path to the ground truth shapefile (e.g., validation points)
    - pred_shp_path: Path to the predicted shapefile (e.g., classification polygons)
    - truth_col: Name of the column containing the true class labels
    - pred_col: Name of the column containing the predicted class labels
    """
    
    # 1. Load the shapefiles
    print("Loading shapefiles...")
    gdf_truth = gpd.read_file(truth_shp_path)
    gdf_pred = gpd.read_file(pred_shp_path)
    
    # 2. Ensure both use the same Coordinate Reference System (CRS)
    if gdf_truth.crs != gdf_pred.crs:
        print(f"CRS mismatch. Reprojecting predictions to {gdf_truth.crs}...")
        gdf_pred = gdf_pred.to_crs(gdf_truth.crs)
        
    # 3. Perform a spatial join
    # This matches truth geometries (like points) to the predicted geometries (like polygons) they intersect
    print("Performing spatial join...")
    joined_gdf = gpd.sjoin(gdf_truth, gdf_pred, how="inner", predicate="intersects")
    
    # Check if the join produced results
    if joined_gdf.empty:
        raise ValueError("Spatial join resulted in an empty dataframe. Ensure the shapefiles geographically overlap.")
    
    # 4. Extract the true and predicted columns
    y_true = joined_gdf[truth_col].astype(str)
    y_pred = joined_gdf[pred_col].astype(str)
    
    # 5. Determine unique class names across both sets
    # Sorting ensures consistent matrix ordering
    class_names = sorted(list(set(y_true.unique()).union(set(y_pred.unique()))))
    
    print(f"Found {len(joined_gdf)} overlapping features. Generating plot...")
    
    # 6. Plot the confusion matrix
    plot_confusion_matrix(y_true, y_pred, class_names)

# ==========================================
# Example Usage:
# ==========================================
if __name__ == "__main__":
    # Define paths to your shapefiles
    GROUND_TRUTH_SHP = "path/to/validation_points.shp"
    CLASSIFICATION_SHP = "path/to/classification_polygons.shp"
    
    # Define the column names containing the class labels/attributes
    TRUTH_LABEL_COLUMN = "LandCover"  # Column in GROUND_TRUTH_SHP
    PRED_LABEL_COLUMN = "Class_ID"    # Column in CLASSIFICATION_SHP
    
    # Run the function
    # shapefiles_to_confusion_matrix(GROUND_TRUTH_SHP, CLASSIFICATION_SHP, TRUTH_LABEL_COLUMN, PRED_LABEL_COLUMN)
```

### Key Considerations for Spatial Data:
1. **Geometry Types**: This template uses `predicate="intersects"`. It works best when `truth_shp_path` is a point shapefile (validation points) and `pred_shp_path` is a polygon shapefile (classification map). If both are polygons, `intersects` will match anywhere their borders touch or overlap, which might create multiple rows for a single polygon.
2. **CRS Checking**: The code automatically checks the Coordinate Reference System (CRS) and reprojects the predicted shapefile to match the ground truth shapefile if they differ. Spatial joins will fail or be wildly inaccurate if the CRSs do not match.
3. **Missing Data**: The `how="inner"` argument ensures that only features that physically overlap and exist in both shapefiles are included in the matrix.


>> User:
confusion matrix for overlapping vs non-overlapping


>> ChatGSFC:
When evaluating "overlapping vs non-overlapping" between two shapefiles (e.g., comparing predicted flooded areas vs. ground truth flooded areas), a standard point-by-point confusion matrix doesn't work well. Instead, you need an **Area-Based Confusion Matrix**.

In this context:
*   **True Positives (TP):** Area where the two shapefiles overlap.
*   **False Positives (FP):** Area in the Prediction shapefile but NOT in the Truth shapefile (Over-prediction).
*   **False Negatives (FN):** Area in the Truth shapefile but NOT in the Prediction shapefile (Missed detection).
*   **True Negatives (TN):** Background area where neither shapefile exists (requires defining a "study area" or bounding box).

Here is a Python template using `geopandas` and `shapely` to calculate and plot an area-based confusion matrix.

### Python Template

```python
import geopandas as gpd
import matplotlib.pyplot as plt
import seaborn as sns
import numpy as np

def plot_area_confusion_matrix(truth_shp_path, pred_shp_path, study_area_shp_path=None):
    """
    Calculates and plots an area-based confusion matrix for overlapping vs non-overlapping shapefiles.
    
    Parameters:
    - truth_shp_path: Path to the ground truth shapefile (polygons)
    - pred_shp_path: Path to the predicted shapefile (polygons)
    - study_area_shp_path: Optional path to a study area/mask shapefile. 
                           If None, uses the bounding box of both shapefiles combined.
    """
    print("Loading shapefiles...")
    gdf_truth = gpd.read_file(truth_shp_path)
    gdf_pred = gpd.read_file(pred_shp_path)
    
    # 1. Ensure matching CRS (Coordinate Reference System)
    if gdf_truth.crs != gdf_pred.crs:
        print(f"Reprojecting predictions to {gdf_truth.crs}...")
        gdf_pred = gdf_pred.to_crs(gdf_truth.crs)
        
    # Warn if using a geographic CRS (like WGS84 / EPSG:4326) instead of projected
    if gdf_truth.crs and gdf_truth.crs.is_geographic:
        print("WARNING: Using a geographic CRS (degrees). Area calculations will be in square degrees. "
              "Consider reprojecting to a local projected CRS (meters) for accurate physical areas.")

    # 2. Merge all geometries into single unified multipolygons
    print("Dissolving geometries...")
    truth_geom = gdf_truth.unary_union
    pred_geom = gdf_pred.unary_union

    # 3. Define the study area (Background)
    if study_area_shp_path:
        gdf_study = gpd.read_file(study_area_shp_path).to_crs(gdf_truth.crs)
        study_geom = gdf_study.unary_union
    else:
        # Fallback: Use the rectangular bounding box of both shapefiles combined
        study_geom = truth_geom.union(pred_geom).envelope

    # 4. Perform spatial operations to calculate areas
    print("Calculating spatial overlaps (areas)...")
    
    # True Positive (Overlap)
    tp_area = truth_geom.intersection(pred_geom).area
    
    # False Negative (Truth but no Prediction)
    fn_area = truth_geom.difference(pred_geom).area
    
    # False Positive (Prediction but no Truth)
    fp_area = pred_geom.difference(truth_geom).area
    
    # True Negative (Inside study area, but neither truth nor prediction)
    union_geom = truth_geom.union(pred_geom)
    tn_area = study_geom.difference(union_geom).area

    # 5. Format into a 2x2 Confusion Matrix array
    # Layout:
    # [[ TN , FP ],
    #  [ FN , TP ]]
    cm_area = np.array([
        [tn_area, fp_area],
        [fn_area, tp_area]
    ])

    # 6. Plotting
    plt.figure(figsize=(8, 6))
    
    # Define labels
    class_names = ['Non-Overlapping (0)', 'Overlapping (1)']
    
    # Create heatmap (using formatting for floats since areas can be decimals)
    sns.heatmap(cm_area, annot=True, fmt='.2f', cmap='Blues', cbar=True,
                xticklabels=class_names, yticklabels=class_names)
    
    plt.title('Area-Based Confusion Matrix\n(Spatial Overlap)', fontsize=16, pad=15)
    plt.ylabel('Ground Truth', fontsize=12, fontweight='bold')
    plt.xlabel('Prediction', fontsize=12, fontweight='bold')
    
    # Add textual interpretation below the plot
    textstr = (f"True Positives (Overlap): {tp_area:.2f}\n"
               f"False Positives (Over-predict): {fp_area:.2f}\n"
               f"False Negatives (Missed): {fn_area:.2f}\n"
               f"True Negatives (Background): {tn_area:.2f}")
    plt.figtext(0.5, -0.1, textstr, wrap=True, horizontalalignment='center', fontsize=11)
    
    plt.tight_layout()
    plt.show()

# ==========================================
# Example Usage:
# ==========================================
if __name__ == "__main__":
    GROUND_TRUTH_SHP = "path/to/truth_polygons.shp"
    PREDICTED_SHP = "path/to/predicted_polygons.shp"
    
    # Optional: A shapefile defining the exact boundary of your analysis
    # STUDY_AREA_SHP = "path/to/study_area_boundary.shp" 
    
    # plot_area_confusion_matrix(GROUND_TRUTH_SHP, PREDICTED_SHP)
```

### Important Notes for this implementation:
1. **Coordinate System (CRS)**: Because this calculates area, you **must** use a projected coordinate system (e.g., UTM) where units are in meters. If your shapefiles are in WGS84 (EPSG:4326), the area will be calculated in square degrees, which distorts the final proportions. 
2. **True Negatives Dependency**: In spatial tasks, the True Negative area is heavily dependent on your defined study area. If you do not provide a `study_area_shp_path`, the script simply draws a bounding box around all your features. If you only care about TP, FP, and FN (Precision and Recall), you can ignore the TN value.

```python
plt.ylabel('Ground Truth', fontsize=12, fontweight='bold')
```

```python
plt.tight_layout()

# --- SAVE THE FIGURE BEFORE plt.show() ---
# bbox_inches='tight' ensures the labels/colorbar don't get cut off
# dpi=300 ensures publication-quality high resolution
plt.savefig('confusion_matrix.png', dpi=300, bbox_inches='tight')

# Optional: You can also save as PDF or SVG for vector graphics

```

### Key Parameters for `savefig`:
*   **`'filename.png'`**: The path and name of the file. You can specify a full path (e.g., `'C:/Users/name/Desktop/confusion_matrix.png'`).
*   **`dpi=300`**: Dots Per Inch. `300` is the standard for high-quality, print-ready images.
*   **`bbox_inches='tight'`**: Highly recommended. It forces matplotlib to calculate the exact bounding box of all elements (like your colorbar label and axis titles) so nothing is cropped out in the saved image.

Here is a complete template using **`rasterio`** and **`geopandas`** to handle the raster-vector comparison, integrated with the styled confusion matrix plot we built previously.

### Prerequisites

```bash
pip install rasterio geopandas numpy matplotlib seaborn
```

### Python Script

```python
import geopandas as gpd
import rasterio
from rasterio import features
import numpy as np
import matplotlib.subplots as plt
import matplotlib.pyplot as plt
import seaborn as sns

def plot_raster_vector_confusion_matrix(truth_shp_path, pred_raster_path):
    """
    Compares a truth shapefile with a prediction raster (1=Flood, 0=Not Flood)
    and plots an area-based confusion matrix.
    """
    print("Loading data...")
    
    # 1. Read the Predicted Raster
    with rasterio.open(pred_raster_path) as src:
        pred_array = src.read(1)  # Read first band
        raster_crs = src.crs
        transform = src.transform
        shape = pred_array.shape
        nodata = src.nodata
        
        # Calculate area of a single pixel (width * height)
        # Assumes a projected CRS (like UTM) where units are meters
        pixel_width = abs(transform[0])
        pixel_height = abs(transform[4])
        pixel_area = pixel_width * pixel_height
        
        if src.crs.is_geographic:
            print("WARNING: Raster is in a geographic CRS (e.g., EPSG:4326). "
                  "Pixel area will be in square degrees, not square meters.")

    # 2. Read and align the Ground Truth Shapefile
    gdf_truth = gpd.read_file(truth_shp_path)
    
    if gdf_truth.crs != raster_crs:
        print(f"Reprojecting shapefile to match raster CRS ({raster_crs})...")
        gdf_truth = gdf_truth.to_crs(raster_crs)

    # 3. Rasterize the Ground Truth shapefile to match the predicted raster
    print("Rasterizing ground truth shapefile...")
    # Extract geometries to burn into the raster
    geometries = [geom for geom in gdf_truth.geometry if geom.is_valid]
    
    if geometries:
        truth_array = features.rasterize(
            geometries,
            out_shape=shape,
            transform=transform,
            fill=0,       # Background is 0
            default_value=1, # Polygons become 1
            dtype=np.uint8
        )
    else:
        # Fallback if shapefile is empty
        truth_array = np.zeros(shape, dtype=np.uint8)

    # 4. Handle NoData regions (Optional but recommended)
    # We only want to evaluate areas where the raster has valid data
    if nodata is not None:
        valid_mask = (pred_array != nodata)
    else:
        # If no NoData value is set, assume all pixels are valid, or map standard no-data like NaN/-9999
        valid_mask = np.ones(shape, dtype=bool)

    # 5. Calculate Pixel Counts for the Confusion Matrix
    print("Calculating overlaps...")
    
    # Boolean masks for True and Predicted flood
    pred_flood = (pred_array == 1) & valid_mask
    truth_flood = (truth_array == 1) & valid_mask
    
    pred_not_flood = (pred_array == 0) & valid_mask
    truth_not_flood = (truth_array == 0) & valid_mask

    # Calculate conditions
    tp_pixels = np.sum(pred_flood & truth_flood)
    fp_pixels = np.sum(pred_flood & truth_not_flood)
    fn_pixels = np.sum(pred_not_flood & truth_flood)
    tn_pixels = np.sum(pred_not_flood & truth_not_flood)

    # Convert pixels to actual area
    tp_area = tp_pixels * pixel_area
    fp_area = fp_pixels * pixel_area
    fn_area = fn_pixels * pixel_area
    tn_area = tn_pixels * pixel_area

    # Format into a 2x2 Confusion Matrix array
    cm_area = np.array([
        [tn_area, fp_area],
        [fn_area, tp_area]
    ])

    # 6. Plot the Confusion Matrix
    class_names = ['Non-Flood (0)', 'Flood (1)']
    
    plt.figure(figsize=(8, 6))
    
    ax = sns.heatmap(cm_area, 
                     annot=True, 
                     fmt='.2f', 
                     cmap='Blues', 
                     cbar=True,
                     xticklabels=class_names, 
                     yticklabels=class_names,
                     annot_kws={"size": 16, "weight": "bold"}) 
    
    # Customize Colorbar
    cbar = ax.collections[0].colorbar
    cbar.set_label('Area (Square Units)', size=16, weight='bold')
    cbar.ax.tick_params(labelsize=12)

    # Customize Labels and Titles
    plt.title('Area-Based Confusion Matrix (Raster vs Vector)', fontsize=18, pad=15)
    plt.ylabel('Ground Truth (Shapefile)', fontsize=14, fontweight='bold')
    plt.xlabel('Prediction (Raster)', fontsize=14, fontweight='bold')
    plt.xticks(fontsize=14)
    plt.yticks(fontsize=14, rotation=0)
    
    plt.tight_layout()
    plt.savefig('raster_vector_confusion_matrix.png', dpi=300, bbox_inches='tight')
    plt.show()
```

### How this works:
1. **Reads Raster Metadata:** It extracts the spatial transform and resolution (`pixel_width` and `pixel_height`) from your predicted raster.
2. **Rasterizes the Shapefile:** Instead of converting millions of raster pixels into polygons (which is very slow), it uses `rasterio.features.rasterize()` to temporarily burn your ground truth polygons into an array that perfectly matches your raster's grid.
3. **NoData Handling:** It looks at the `.nodata` attribute of your raster to exclude edges or unmapped areas from skewing your "True Negative" count.
4. **Area Conversion:** It counts the overlapping pixels and multiplies them by the `pixel_area` to give you the results in square meters (assuming your data is in a metric projected CRS, like UTM).
