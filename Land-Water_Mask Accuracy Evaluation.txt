Here is a Python script using the `rasterio` library to read the GeoTIFFs and `scikit-learn` to calculate standard classification metrics (Overall Accuracy, Precision, Recall, F1-Score, and a Confusion Matrix).

```python
import rasterio
import numpy as np
from sklearn.metrics import accuracy_score, precision_recall_fscore_support, confusion_matrix, classification_report

def evaluate_mask_accuracy(predicted_tiff, reference_tiff, nodata_value=None):
    """
    Evaluates the accuracy of a land-water mask against a reference mask.
    
    Parameters:
    - predicted_tiff: str, path to the classified/predicted land-water mask GeoTIFF
    - reference_tiff: str, path to the ground truth/reference land-water mask GeoTIFF
    - nodata_value: int/float, value to ignore during evaluation (e.g., 255)
    """
    
    # Open both GeoTIFFs
    with rasterio.open(predicted_tiff) as src_pred, rasterio.open(reference_tiff) as src_ref:
        
        # Check if dimensions match
        if src_pred.shape != src_ref.shape:
            raise ValueError(f"Shape mismatch: Predicted {src_pred.shape} vs Reference {src_ref.shape}. "
                             f"Rasters must be aligned and have the same dimensions.")
        
        # Read the first band of the raster into numpy arrays
        pred_data = src_pred.read(1)
        ref_data = src_ref.read(1)
        
        # Flatten arrays for scikit-learn metrics
        pred_flat = pred_data.flatten()
        ref_flat = ref_data.flatten()
        
        # Handle NoData values if specified or available in raster metadata
        ref_nodata = nodata_value if nodata_value is not None else src_ref.nodata
        pred_nodata = nodata_value if nodata_value is not None else src_pred.nodata
        
        # Create a valid data mask (ignoring NoData pixels in either raster)
        valid_mask = np.ones_like(ref_flat, dtype=bool)
        if ref_nodata is not None:
            valid_mask &= (ref_flat != ref_nodata)
        if pred_nodata is not None:
            valid_mask &= (pred_flat != pred_nodata)
            
        # Filter the arrays using the valid mask
        y_true = ref_flat[valid_mask]
        y_pred = pred_flat[valid_mask]
        
        # Calculate metrics
        print("--- Land-Water Mask Evaluation ---")
        
        # 1. Overall Accuracy
        oa = accuracy_score(y_true, y_pred)
        print(f"Overall Accuracy: {oa:.4f}")
        
        # 2. Confusion Matrix
        cm = confusion_matrix(y_true, y_pred)
        print("\nConfusion Matrix:")
        print(cm)
        
        # 3. Detailed Classification Report (Precision, Recall, F1-Score)
        # Assuming 0 = Water, 1 = Land (adjust labels as needed for your specific data)
        print("\nClassification Report:")
        print(classification_report(y_true, y_pred, digits=4))

### Important Considerations:
1. **Alignment & Projection:** This code assumes that both GeoTIFFs have exactly the same dimensions (rows and columns), spatial resolution, and coordinate reference system (CRS). If they do not match, you will need to reproject or resample one to match the other using `rasterio.warp` or a tool like `gdalwarp` before running this script.
2. **Class Values:** Make sure both rasters use the same integer mapping (e.g., `0` for Water, `1` for Land). If your predicted mask uses `1` and `2`, and the reference uses `0` and `1`, you will need to map/reclassify the values using `numpy` before passing them to the metric functions.
3. **NoData Values:** Areas outside the valid domain (like image edges) will skew your accuracy artificially high if both masks agree on "NoData". The code actively masks out these pixels. Ensure your `nodata_value` parameter is correct.

import rasterio
from rasterio.warp import transform_bounds, reproject, Resampling
import numpy as np
from sklearn.metrics import accuracy_score, confusion_matrix, classification_report

def evaluate_masks_with_reprojection(ref_tiff, pred_tiff, nodata_value=None):
    """
    Reads two Geotiffs of different sizes/projections, extracts the common overlapping
    area, reprojects the predicted mask to match the reference mask, and evaluates accuracy.
    
    Parameters:
    - ref_tiff: str, path to the reference (ground truth) GeoTIFF. 
                This will be the target projection and resolution.
    - pred_tiff: str, path to the predicted mask GeoTIFF.
    - nodata_value: int/float, value to ignore during evaluation.
    """
    
    with rasterio.open(ref_tiff) as src_ref, rasterio.open(pred_tiff) as src_pred:
        
        print(f"Reference CRS: {src_ref.crs}")
        print(f"Predicted CRS: {src_pred.crs}")
        
        # 1. Transform the bounds of the predicted image into the reference CRS
        pred_bounds_in_ref_crs = transform_bounds(
            src_pred.crs, src_ref.crs, *src_pred.bounds
        )
        
        # 2. Calculate the intersection of the two bounding boxes
        ref_bounds = src_ref.bounds
        intersect_left = max(ref_bounds.left, pred_bounds_in_ref_crs[0])
        intersect_bottom = max(ref_bounds.bottom, pred_bounds_in_ref_crs[1])
        intersect_right = min(ref_bounds.right, pred_bounds_in_ref_crs[2])
        intersect_top = min(ref_bounds.top, pred_bounds_in_ref_crs[3])
        
        # Check if they actually overlap
        if intersect_left >= intersect_right or intersect_bottom >= intersect_top:
            raise ValueError("The two GeoTIFFs do not overlap in space.")
            
        print("Calculated intersecting bounding box successfully.")
        
        # 3. Get the pixel window in the reference image corresponding to the overlap
        window = src_ref.window(intersect_left, intersect_bottom, intersect_right, intersect_top)
        
        # Round the window to integer pixel coordinates to ensure exact alignment
        window = window.round_offsets().round_lengths()
        
        # 4. Read the reference data for the overlapping window
        ref_data = src_ref.read(1, window=window)
        
        # Get the spatial transform for this specific window
        window_transform = src_ref.window_transform(window)
        
        # 5. Create an empty array to hold the reprojected predicted data
        # It must exactly match the shape of the extracted reference window
        pred_data = np.empty_like(ref_data)
        
        # 6. Reproject and resample the predicted image into the new array
        # We use NEAREST resampling because land/water masks are categorical
        print("Reprojecting and resampling predicted image to match reference grid...")
        reproject(
            source=rasterio.band(src_pred, 1),
            destination=pred_data,
            src_transform=src_pred.transform,
            src_crs=src_pred.crs,
            dst_transform=window_transform,
            dst_crs=src_ref.crs,
            resampling=Resampling.nearest 
        )
        
        # --- Evaluate Accuracy ---
        
        # Flatten arrays for scikit-learn
        ref_flat = ref_data.flatten()
        pred_flat = pred_data.flatten()
        
        # Identify NoData values
        ref_nodata = nodata_value if nodata_value is not None else src_ref.nodata
        pred_nodata = nodata_value if nodata_value is not None else src_pred.nodata
        
        # Create a valid data mask (ignore NoData)
        valid_mask = np.ones_like(ref_flat, dtype=bool)
        if ref_nodata is not None:
            valid_mask &= (ref_flat != ref_nodata)
        if pred_nodata is not None:
            valid_mask &= (pred_flat != pred_nodata)
            
        # Filter arrays
        y_true = ref_flat[valid_mask]
        y_pred = pred_flat[valid_mask]
        
        if len(y_true) == 0:
            print("No valid pixels to evaluate after applying NoData masks.")
            return

        print("\n--- Land-Water Mask Evaluation (Overlapping Area) ---")
        print(f"Evaluated Pixels: {len(y_true)}")
        
        # Overall Accuracy
        oa = accuracy_score(y_true, y_pred)
        print(f"Overall Accuracy: {oa:.4f}")
        
        # Confusion Matrix
        cm = confusion_matrix(y_true, y_pred)
        print("\nConfusion Matrix:")
        print(cm)
        
        # Classification Report
        print("\nClassification Report:")
        print(classification_report(y_true, y_pred, digits=4))

# Example usage
if __name__ == "__main__":
    # The reference image acts as the base map (target CRS and target grid)
    reference_file = "path/to/ground_truth.tif"
    
    # The predicted image will be reprojected, cropped, and resampled to match the reference
    predicted_file = "path/to/model_prediction.tif"
    
    # Run evaluation (assumes 255 represents NoData)
    evaluate_masks_with_reprojection(reference_file, predicted_file, nodata_value=255)
```

### How this works:
1. **`transform_bounds`**: It calculates the geographic footprint of Image 2 but expresses those coordinates in the EPSG system of Image 1.
2. **Intersection Calculation**: It finds the mathematical overlap (bounding box) of both footprints.
3. **`src_ref.window`**: It converts those geographic bounding box coordinates into pixel coordinates (row/col) specific to Image 1's grid.
4. **`reproject`**: Instead of saving a new physical file to disk, it performs an in-memory transformation. It projects the raster from Image 2 directly into an empty Numpy array that perfectly matches the height, width, and geographic location of the Image 1 window.

import numpy as np

def evaluate_land_accuracy(ref_tiff, pred_tiff, nodata_value=None):

    Evaluates land mask accuracy where NoData represents Water and valid data represents Land.
    """

            source=rasterio.band(src_pred, 1),
            destination=pred_data,
            src_transform=src_pred.transform,
            src_crs=src_pred.crs,
            dst_transform=window_transform,
            dst_crs=src_ref.crs,
            resampling=Resampling.nearest 
        )
        
        # --- Evaluate Land Accuracy ---
        ref_flat = ref_data.flatten()
        pred_flat = pred_data.flatten()
        
        ref_nodata = nodata_value if nodata_value is not None else src_ref.nodata
        pred_nodata = nodata_value if nodata_value is not None else src_pred.nodata
        
        if ref_nodata is None or pred_nodata is None:
            raise ValueError("NoData values must be defined in the raster metadata or passed as an argument.")

        actual_land = (ref_flat != ref_nodata)
        predicted_land = (pred_flat != pred_nodata)

        f1_score = 2 * (precision * recall) / (precision + recall) if (precision + recall) > 0 else 0.0
        iou = true_positives / (true_positives + false_positives + false_negatives) if (true_positives + false_positives + false_negatives) > 0 else 0.0

import os
import glob
import rasterio
from rasterio.warp import reproject, Resampling
import numpy as np

def evaluate_directory_composite(ref_tiff, pred_dir, ref_nodata_value=None, pred_nodata_value=None):
    """
    Evaluates all predicted TIFFs in a directory against a single large reference TIFF.
    Calculates composite Land accuracy statistics across all images.
    """
    
    # Find all TIFF files in the predicted directory
    pred_files = glob.glob(os.path.join(pred_dir, "*.tif"))
    
    if not pred_files:
        print(f"No .tif files found in directory: {pred_dir}")
        return

    print(f"Found {len(pred_files)} images to process. Calculating composite statistics...\n")

    # Initialize running totals
    total_tp = 0
    total_fp = 0
    total_fn = 0
    total_actual = 0
    total_predicted = 0

    # Open the large reference image once
    with rasterio.open(ref_tiff) as src_ref:
        ref_nodata = ref_nodata_value if ref_nodata_value is not None else src_ref.nodata
        
        if ref_nodata is None:
             raise ValueError("Reference NoData value must be defined.")

        for idx, pred_file in enumerate(pred_files, 1):
            print(f"Processing {idx}/{len(pred_files)}: {os.path.basename(pred_file)}")
            
            with rasterio.open(pred_file) as src_pred:
                pred_nodata = pred_nodata_value if pred_nodata_value is not None else src_pred.nodata
                
                if pred_nodata is None:
                    print(f"  Skipping {os.path.basename(pred_file)}: NoData value undefined.")
                    continue
                
                # Read predicted data
                pred_data = src_pred.read(1)
                
                # Create empty array for reprojected reference data
                ref_data_reproj = np.empty_like(pred_data)
                
                # Reproject reference to match the current predicted image
                reproject(
                    source=rasterio.band(src_ref, 1),
                    destination=ref_data_reproj,
                    src_transform=src_ref.transform,
                    src_crs=src_ref.crs,
                    dst_transform=src_pred.transform,
                    dst_crs=src_pred.crs,
                    resampling=Resampling.nearest 
                )
                
                # Flatten arrays
                pred_flat = pred_data.flatten()
                ref_flat = ref_data_reproj.flatten()
                
                # Define Land as True (Not NoData) and Water as False (NoData)
                actual_land = (ref_flat != ref_nodata)
                predicted_land = (pred_flat != pred_nodata)
                
                # Accumulate counts for this image
                total_actual += np.sum(actual_land)
                total_predicted += np.sum(predicted_land)
                
                total_tp += np.sum(actual_land & predicted_land)
                total_fp += np.sum(~actual_land & predicted_land)
                total_fn += np.sum(actual_land & ~predicted_land)

    # --- Calculate Composite Metrics ---
    print("\n" + "="*50)
    print("      COMPOSITE LAND ACCURACY REPORT")
    print("="*50)
    
    if total_actual == 0 and total_predicted == 0:
        print("No land pixels found in any of the images.")
        return

    # Calculate final metrics using aggregated counts
    recall = total_tp / total_actual if total_actual > 0 else 0.0
    precision = total_tp / total_predicted if total_predicted > 0 else 0.0
    f1_score = 2 * (precision * recall) / (precision + recall) if (precision + recall) > 0 else 0.0
    iou = total_tp / (total_tp + total_fp + total_fn) if (total_tp + total_fp + total_fn) > 0 else 0.0
    
    print(f"Total Images Processed:      {len(pred_files)}")
    print(f"Total Actual Land Pixels:    {total_actual:,}")
    print(f"Total Predicted Land Pixels: {total_predicted:,}")
    print(f"Total Correctly Predicted:   {total_tp:,}")
    print("-" * 50)
    print(f"Composite Recall (Producer's): {recall:.4f}")
    print(f"Composite Precision (User's):  {precision:.4f}")
    print(f"Composite F1-Score:            {f1_score:.4f}")
    print(f"Composite IoU:                 {iou:.4f}")
    print("="*50)
