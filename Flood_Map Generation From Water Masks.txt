```python
import rasterio
import numpy as np
import os

def generate_flood_map(reference_mask_path, event_mask_path, output_flood_path, nodata_value=255):
    """
    Generates a flood map from reference and event water masks.
    Assumes: 0 = Land, 1 = Water
    Flood condition: Event is Water (1) AND Reference is Land (0)
    """
    print(f"Loading reference mask: {reference_mask_path}")
    with rasterio.open(reference_mask_path) as ref_src:
        ref_data = ref_src.read(1)
        # Copy metadata for the output file
        out_meta = ref_src.meta.copy()
        
    print(f"Loading event mask: {event_mask_path}")
    with rasterio.open(event_mask_path) as event_src:
        event_data = event_src.read(1)
        
    # Check if spatial dimensions match
    if ref_data.shape != event_data.shape:
        raise ValueError(f"Raster shape mismatch! Reference: {ref_data.shape}, Event: {event_data.shape}")

    print("Calculating flood extent...")
    # Calculate flood mask using bitwise AND: (event == 1) & (reference == 0)
    # This results in a boolean array, which we convert to an 8-bit integer
    flood_map = np.where((event_data == 1) & (ref_data == 0), 1, 0).astype(np.uint8)
    
    # Optional: If you have a nodata mask in your original files, apply it to the output
    # For example, if NoData is 255 in the input files:
    # flood_map = np.where((event_data == 255) | (ref_data == 255), nodata_value, flood_map)

    # Update metadata to ensure output is written correctly
    out_meta.update({
        'dtype': 'uint8',
        'count': 1,
        'nodata': nodata_value,
        'compress': 'lzw' # LZW compression to keep file size small
    })

    print(f"Saving flood map to: {output_flood_path}")
    # Write the calculated flood mask to a new GeoTIFF
    with rasterio.open(output_flood_path, 'w', **out_meta) as dest:
        dest.write(flood_map, 1)
        
    print("Flood map generation complete!")
```

The standard approach for this in remote sensing is to **reproject and resample** the Reference imagery to match the exact spatial footprint, resolution, and coordinate reference system of the Event imagery.

```python
import os

def align_raster_to_target(source_path, target_shape, target_transform, target_crs):
    """Helper function to read and align a source raster to a target grid."""
    with rasterio.open(source_path) as src:
        # Create an empty array with the exact shape of the target
        aligned_data = np.zeros(target_shape, dtype=src.dtypes[0])
        
        # Reproject/Resample the source data into the target grid
        reproject(
            source=rasterio.band(src, 1),
            destination=aligned_data,
            src_transform=src.transform,
            src_crs=src.crs,
            dst_transform=target_transform,
            dst_crs=target_crs,
            resampling=Resampling.nearest  # Crucial for categorical masks
        )
        return aligned_data

def generate_flood_map_with_lc(reference_path, event_path, landcover_path, output_path, land_classes, nodata_value=255):
    """
    Generates a flood map using reference water, event water, and a land cover constraint.
    
    Conditions for flooding:
    1. Event imagery shows water (1)
    2. Reference imagery shows land (0)
    3. Land Cover map class is in the accepted `land_classes` list
    """
    
    print(f"Loading event mask (target grid): {event_path}")
    with rasterio.open(event_path) as event_src:
        event_data = event_src.read(1)
        event_meta = event_src.meta.copy()
        event_transform = event_src.transform
        event_crs = event_src.crs
        
    print(f"Loading and aligning reference mask: {reference_path}")
    ref_aligned = align_raster_to_target(reference_path, event_data.shape, event_transform, event_crs)
    
    print(f"Loading and aligning land cover map: {landcover_path}")
    lc_aligned = align_raster_to_target(landcover_path, event_data.shape, event_transform, event_crs)

    print("Calculating flood extent...")
    
    # Define boolean conditions
    is_event_water = (event_data == 1)
    is_ref_land = (ref_aligned == 0)
    
    # Check if the land cover pixel is inside the provided list of valid land classes
    # np.isin checks every pixel in lc_aligned against the list of land_classes
    is_lc_land = np.isin(lc_aligned, land_classes)
    
    # A pixel is flooded ONLY if all three conditions are True
    flood_map = np.where(is_event_water & is_ref_land & is_lc_land, 1, 0).astype(np.uint8)
    
    # Apply NoData masking if any of the inputs contained the NoData value in that pixel
    any_nodata = (event_data == nodata_value) | (ref_aligned == nodata_value) | (lc_aligned == nodata_value)
    flood_map = np.where(any_nodata, nodata_value, flood_map)

    event_meta.update({
        'dtype': 'uint8',
        'count': 1,
        'nodata': nodata_value,
        'compress': 'lzw'
    })

    print(f"Saving aligned flood map to: {output_path}")
    with rasterio.open(output_path, 'w', **event_meta) as dest:
        dest.write(flood_map, 1)
        
    print("Flood map generation complete!")
```

```python
import rasterio
from rasterio.warp import reproject, Resampling
from rasterio import features
import geopandas as gpd
import numpy as np
import os

def align_raster_to_target(source_path, target_shape, target_transform, target_crs):
    """Reads and aligns a source raster to a target grid."""
    with rasterio.open(source_path) as src:
        aligned_data = np.zeros(target_shape, dtype=src.dtypes[0])
        reproject(
            source=rasterio.band(src, 1),
            destination=aligned_data,
            src_transform=src.transform,
            src_crs=src.crs,
            dst_transform=target_transform,
            dst_crs=target_crs,
            resampling=Resampling.nearest
        )
        return aligned_data

def generate_flood_map_vector_lc(reference_path, event_path, vector_lc_path, output_path, nodata_value=255):
    """
    Generates a flood map using reference water, event water, and a vector land mask.
    
    Conditions for flooding:
    1. Event imagery shows water (1)
    2. Reference imagery shows land (0)
    3. Pixel falls inside a vector feature from the shapefile
    """
    
    print(f"Loading event mask (target grid): {event_path}")
    with rasterio.open(event_path) as event_src:
        event_data = event_src.read(1)
        event_meta = event_src.meta.copy()
        event_transform = event_src.transform
        event_crs = event_src.crs
        event_shape = event_data.shape
        
    print(f"Loading and aligning reference raster mask: {reference_path}")
    ref_aligned = align_raster_to_target(reference_path, event_shape, event_transform, event_crs)
    
    print(f"Loading and rasterizing vector land mask: {vector_lc_path}")
    # Load the shapefile
    gdf = gpd.read_file(vector_path)
    
    # Reproject the shapefile to match the Event raster's CRS if they differ
    if gdf.crs != event_crs:
        print(f"Reprojecting vector from {gdf.crs} to {event_crs}...")
        gdf = gdf.to_crs(event_crs)
    
    # Rasterize the geometries into a boolean mask (1 = inside feature, 0 = outside)
    if len(gdf) > 0:
        lc_mask = features.rasterize(
            shapes=gdf.geometry,
            out_shape=event_shape,
            transform=event_transform,
            fill=0,          # Background value (not land)
            default_value=1, # Value assigned to areas inside polygons (land)
            dtype=np.uint8
        )
    else:
        print("Warning: Vector file is empty. No land detected.")
        lc_mask = np.zeros(event_shape, dtype=np.uint8)

    print("Calculating flood extent...")
    
    # Define boolean conditions
    is_event_water = (event_data == 1)
    is_ref_land = (ref_aligned == 0)
    is_vector_land = (lc_mask == 1)
    
    # A pixel is flooded ONLY if all three conditions are True
    flood_map = np.where(is_event_water & is_ref_land & is_vector_land, 1, 0).astype(np.uint8)
    
    # Apply NoData masking if either of the raster inputs contained the NoData value
    any_nodata = (event_data == nodata_value) | (ref_aligned == nodata_value)
    flood_map = np.where(any_nodata, nodata_value, flood_map)

    # Update metadata for output
    event_meta.update({
        'dtype': 'uint8',
        'count': 1,
        'nodata': nodata_value,
        'compress': 'lzw'
    })

    print(f"Saving flood map to: {output_path}")
    with rasterio.open(output_path, 'w', **event_meta) as dest:
        dest.write(flood_map, 1)
        
    print("Flood map generation complete!")
```
